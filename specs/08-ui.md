## JavaScript Implementation

### State Management

```javascript
let currentTab = 'drafts';  // 'drafts', 'confirmed', 'all'
let expenses = [];
let stats = {};
```

### Key Functions

```javascript
// Load expenses from API
async function loadExpenses() {
    const params = currentTab !== 'all' ? `?status=${currentTab}` : '';
    const response = await fetch(`/api/expenses${params}`);
    expenses = await response.json();
    renderExpenses();
}

// Load stats from API
async function loadStats() {
    const response = await fetch('/api/stats');
    stats = await response.json();
    renderStats();
}

// Parse text with AI
async function parseText(text, subject) {
    showLoading();
    const response = await fetch('/api/parse-text', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ text, subject })
    });
    const result = await response.json();
    hideLoading();
    if (result.success) {
        openExpenseForm(result.data);
    } else {
        showError(result.error);
    }
}

// Parse PDF with AI
async function parsePDF(file) {
    showLoading();
    const formData = new FormData();
    formData.append('file', file);
    const response = await fetch('/api/parse-pdf', {
        method: 'POST',
        body: formData
    });
    const result = await response.json();
    hideLoading();
    if (result.success) {
        openExpenseForm(result.data, result.filename, result.file_data);
    } else {
        showError(result.error);
    }
}
```

### Auto-Refresh

```javascript
// Refresh stats and expenses every 30 seconds
setInterval(() => {
    loadStats();
    loadExpenses();
}, 30000);
```
